2) Update useLibraryEntries to accept division scope
Replace current hook:

function useLibraryEntries(type: string) {
  return useQuery<LibraryEntry[]>({
    queryKey: ["/api/library", type],
    queryFn: async () => {
      const res = await fetch(`/api/library?type=${type}`);
      if (!res.ok) throw new Error("Failed to fetch");
      return res.json();
    },
  });
}

With:

function useLibraryEntries(type: string, divisionCode?: string | null) {
  const divKey = divisionCode ?? "all";
  return useQuery<LibraryEntry[]>({
    queryKey: ["/api/library", type, divKey],
    queryFn: async () => {
      const url = `/api/library?type=${encodeURIComponent(type)}${divisionCode ? `&divisionCode=${encodeURIComponent(divisionCode)}` : ""}`;
      const res = await fetch(url);
      if (!res.ok) throw new Error("Failed to fetch");
      return res.json();
    },
  });
}
